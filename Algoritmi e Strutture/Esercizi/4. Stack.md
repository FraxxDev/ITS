```python
    VARIABILI:
        TempStack: Pila  // Pila di supporto
        elemento: TipoDiDato

    INIZIO

        // Inizializza la pila temporanea vuota
        Crea(TempStack)

        // FASE 1: Filtro e travaso
        // Svuotiamo S spostando in TempStack solo gli elementi diversi da k
        
        MENTRE (S non è vuota) FARE
            elemento ← S.pop()  // Estrai l'elemento in cima a S
            
            SE (elemento ≠ k) ALLORA
                TempStack.push(elemento) // Salvalo se non è quello da eliminare
            ALTRIMENTI
                // Se elemento == k, non fare nulla (l'elemento viene scartato)
            FINE SE
        FINE MENTRE

        // A questo punto S è vuota e TempStack contiene gli elementi validi
        // ma in ordine inverso rispetto all'originale.

        // FASE 2: Ripristino ordine
        // Travasiamo tutto da TempStack a S per ripristinare l'ordine originale
        
        MENTRE (TempStack non è vuota) FARE
            elemento ← TempStack.pop()
            S.push(elemento)
        FINE MENTRE

    FINE
```
